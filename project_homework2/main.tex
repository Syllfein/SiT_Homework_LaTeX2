\section*{Вариант 7}

Если требуется улучшить качество переходных процессов при некотором фиксированном $$s = s_0,$$ то в этом случае целесообразно выполнить параметрический синтез, т. е. выбор значений параметров обратных связей, на основе минимизации функции $$f(p, s_0),$$ где

$$
f(p, s) =
\left(
\|R_A(0, p_0, s)\|^{-2} + \|R_A(0, p, s)\|^{-2}
\right)
\int\limits_0^{\infty} f(p, s, \omega)\, d\omega,
$$

$$
f(p, s, \omega) =
\|R_A(\omega, p, s) - R_A(0, p, s)R_A^*(\omega)\|^2 + c_1
\left\|
R_A'(\omega, p, s) - R_A(0, p, s)R_A^{*'}(\omega)
\right\|^2 +
$$

$$
+ c_2
\left\|
R_A''(\omega, p, s) - R_A(0, p, s)R_A^{*''}(\omega)
\right\|^2,
()' = \frac{d()}{d\omega},
$$

$$
R_{A\nu j}(\omega, p, s) =
\begin{cases}
|\Phi_{\nu j}(i\omega, p, s)|, & A_j(p, s) \neq 0, \\
\sqrt{1 + \omega^2\, |\Psi_{\nu j}(i\omega, p, s)|}, & A_j(p, s) = 0.
\end{cases}
$$
